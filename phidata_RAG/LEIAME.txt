# Estudo de Caso - Deploy de Time de Agentes de IA com DeepSeek Para Análise de Inteligência de Mercado

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name aiagentenv python=3.12

# Ative o ambiente:

conda activate aiagentenv (ou: source activate aiagentenv)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Execute a app:

python dsa_agentes.py

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name aiagentenv --all
